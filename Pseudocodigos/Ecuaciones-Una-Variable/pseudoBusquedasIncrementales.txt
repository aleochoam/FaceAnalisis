INICIO

    LEER p_ini, delta, iters, fun 

    fun_p_ini <-- fun(p_ini)
    
    SI fun_p_ini = 0 ENTONCES:
        IMPRIMIR p_ini es raiz
    SINO:
        p_sig <-- p_ini + delta
        cont <-- 1
        fun_p_sig <-- fun(p_sig)

        MIENTRAS fun_p_ini * fun_p_sig > 0 Y cont <= iters HAGA:
            p_ini <-- p_sig
            fun_p_ini <-- fun_p_sig
            p_sig <-- p_ini + delta
            fun_p_sig <-- fun(p_sig)
            cont <-- cont + 1
        FIN_MIENTRAS

        SI fun_p_ini < 0 ENTONCES:
            IMPRIMIR p_ini es raiz
        SINO SI fun_p_ini * fun_p_sig < 0 ENTONCES:
            IMPRIMIR Hay una raiz entre p_ini y p_sig
        SINO
            Fracaso en iters iteraciones
        FIN_SI
    FIN_SI
FIN_INICIO
